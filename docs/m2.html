Chapter 7 M2

Packages.

TROLL simulation an prelog forest.

sim <- troll(name = "pre",
             global = generate_parameters(cols = 300, rows = 300,
                                          iterperyear = 12, nbiter = 12*600),
             species = TROLLv3_species,
             climate = TROLLv3_climatedaytime12,
             daily = TROLLv3_daytimevar,
             verbose = FALSE)
prelog <- get_forest(sim) %>%
  mutate(Forest = "Paracou") %>%
  mutate(Plot = "6") %>%
  mutate(CensusYear = 2022) %>%
  mutate(PlotArea = 4) %>%
  mutate(idTree = as.integer(as.factor(paste(col, row)))) %>%
  mutate(CodeAlive = TRUE) %>%
  tidyr::separate(s_name, c("Genus", "Species"), remove = F) %>%
  left_join(vroom::vroom("save/M2/paracou.tsv") %>%
              select(Family, Genus, Species) %>%
              unique(), by = c("Genus", "Species")) %>%
  mutate(VernName = NA_character_) %>%
  mutate(DBH = dbh*100) %>%
  mutate(Circ = dbh*100*pi, CircCorr = dbh*100*pi) %>%
  mutate(Xfield = col, Yfield = row) %>%
  mutate(Xutm = 286395 + Xfield, Yutm = 582935.9 + Yfield, UTMZone = as.integer(22)) %>%
  mutate(Lat = Xutm, Lon = Yutm)
write_tsv(prelog, "save/M2/prelog.tsv")

prelog <- readr::read_tsv("save/M2/prelog.tsv")
